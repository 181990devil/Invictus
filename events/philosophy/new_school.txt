new_school.2 = { # Find Platonist philosophers

	title = "new_school.2.t"

	right_portrait = scope:potential_platonist_scholarch_1
	right_portrait = scope:potential_platonist_scholarch_2

	picture = great_city
	
	immediate = {
		
		if = {
			limit = {
				any_character = {
					has_character_modifier = platonist_cmod
					NOT = {
						has_character_modifier = scholarch
					}
					# has_any_office = no
					# is_governor = no
					# in_command = no
					# is_admiral = no
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_1}
						NOT = {	this = scope:potential_platonist_scholarch_1}
					}
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_2}
						NOT = {	this = scope:potential_platonist_scholarch_2}
					}
				}
			}
			random_character = {
				limit = {
					has_character_modifier = platonist_cmod
					NOT = {
						has_character_modifier = scholarch
					}
					# has_any_office = no
					# is_governor = no
					# in_command = no
					# is_admiral = no
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_1}
						NOT = {	this = scope:potential_platonist_scholarch_1}
					}
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_2}
						NOT = {	this = scope:potential_platonist_scholarch_2}
					}
				}
				save_scope_as = potential_platonist_scholarch_1
			}
		}
		else = {
			every_country = {
				limit = {
					OR = {
						this = root
						in_diplomatic_range = root
					}
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_1_origin}
						NOT = {	this = scope:potential_platonist_scholarch_1_origin}
					}
				}
				random_character = {
					limit = {
						has_character_modifier = platonist_cmod
						NOT = {
							has_character_modifier = scholarch
						}
						has_any_office = no
						is_governor = no
						in_command = no
						is_admiral = no
						is_ruler = no
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_1}
							NOT = {	this = scope:potential_platonist_scholarch_1}
						}
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_2}
							NOT = {	this = scope:potential_platonist_scholarch_2}
						}
					}
					save_scope_as = potential_platonist_scholarch_1
					scope:potential_platonist_scholarch_1.employer = {
						save_scope_as = potential_platonist_scholarch_1_origin
					}
				}
			}
		}
		
		if = {
			limit = {
				any_country = {
					OR = {
						this = root
						in_diplomatic_range = root
					}
					any_character = {
						has_character_modifier = platonist_cmod
						NOT = {
							has_character_modifier = scholarch
						}
						has_any_office = no
						is_governor = no
						in_command = no
						is_admiral = no
						is_ruler = no
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_1}
							NOT = {	this = scope:potential_platonist_scholarch_1}
						}
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_2}
							NOT = {	this = scope:potential_platonist_scholarch_2}
						}
					}
				}
			}
		
			every_country = {
				limit = {
					OR = {
						this = root
						in_diplomatic_range = root
					}
					trigger_if = {
						limit = {exists = scope:potential_platonist_scholarch_2_origin}
						NOT = {	this = scope:potential_platonist_scholarch_2_origin}
					}
				}
				random_character = {
					limit = {
						has_character_modifier = platonist_cmod
						NOT = {
							has_character_modifier = scholarch
						}
						has_any_office = no
						is_governor = no
						in_command = no
						is_admiral = no
						is_ruler = no
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_1}
							NOT = {	this = scope:potential_platonist_scholarch_1}
						}
						trigger_if = {
							limit = {exists = scope:potential_platonist_scholarch_2}
							NOT = {	this = scope:potential_platonist_scholarch_2}

					}
