flavor_maurya.3 = {		#Convert Changraputa, a few charcters, and the capital to Jain.
    type = country_event

		}

				if = {
					limit = {
						any_pops_in_province = {
							OR = {
							pop_type = nobles
							pop_type = citizen
							pop_type = freemen
							}
						}
					}
					while = {
						count = 25
						limit = {
							any_pops_in_province = {
								OR = {
								pop_type = nobles
								pop_type = citizen
								pop_type = freemen
								}
							}
						}
						random_pops_in_province = {
							limit = {
								OR = {
								pop_type = nobles
								pop_type = citizen
								pop_type = freemen
								}
							}
							set_pop_religion = jainism
						}
					}

flavor_maurya.5 = {		#The death of Bindusara
    type = country_event
    title = flavor_maurya_5_title
    desc = flavor_maurya_5_desc
    picture = aqueducts
    
    left_portrait = char:65
    right_portrait = char:66

    trigger = {	#Bindusara is alive and Ashoka is 20
    	tag = MRY
    	char:65 = {
    		is_alive = yes
    	}
    	char:66 = {
    		age >= 20
    	}
    	NOT = { has_variable = bindusara_is_a_shithead }
    }

    immediate = {	
		char:65 = {
	 		death = {
	 			death_reason = death_murdered_in_plot
			}
		}
		char:66 = {
			c:MRY= {
				set_as_ruler=char:66
			}
		}
		hidden_effect = {
            if = {
                limit = {
                    any_character = {
                        is_adult = yes
                        is_primary_heir = no
                        is_ruler = no
                        is_governor = yes
                    }
                }
                ordered_character = {
                    limit = {
                        is_adult = yes
                        is_primary_heir = no
                        is_ruler = no
                        is_governor = yes                        
                    }
                    check_range_bounds = no
                    order_by = power_base
                    save_scope_as = rebel_scum_leader
                    max = 1
                }
            }
            else = {
                ordered_character = {
                    limit = {
                        is_ruler = no
                        is_adult = yes
                        is_primary_heir = no
                        is_minor_character = yes
                    }
                    max = 1
                    check_range_bounds = no
                    order_by = power_base
                    save_scope_as = rebel_scum_leader
               }
            }
            ordered_governorships = {
                limit = {
                    NOT = {
                        any_governorship_state = {
                            is_capital_state = yes
                        }
                    }
                }
                max = 1
                order_by = {
                    every_governorship_state = {
                        every_state_province = {
                            add = total_population
                        }
                    }
                }
                save_scope_as = rebel_scum_base
            }
            if = {
                limit = {
                    scope:rebel_scum_leader = {
                        is_governor = no
                    }
                }
                scope:rebel_scum_base = {
                    random_governorship_state = {
                        random_state_province = {
                            set_as_governor = scope:rebel_scum_leader
                        }
                    }
                }
            }
            start_civil_war = scope:rebel_scum_leader
            scope:rebel_scum_leader.location = {
                save_scope_as = civil_war_cap
            }
            scope:civil_war_cap.owner = {
                change_country_name = "maurya_revolter_name"
                change_country_adjective = "maurya_revolter_adjective"
                change_country_color = bharat_green_color
                change_country_flag = BHARAT_REVOLT_FLAG
                change_government = imperium
                save_scope_as = bharat_revolt_country
            }
        }		  	
    }

    option = {    #Ashoka will bring greatness to Maurya. (Start a small civil war)
        name = flavor_maurya_5.a

    }
}
