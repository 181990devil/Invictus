me_scythia_01_greek_stance.8 = {							#	Bonds With Olbia
    type = country_event
    title = me_scythia_01_greek_stance.8.t
    desc = me_scythia_01_greek_stance.8.desc
    picture = greek_city
	goto_location = p:4529
    left_portrait = current_ruler
    right_portrait = scope:c:OLB.current_ruler

    trigger = { 
		has_land = yes
		exists = c:OLB
 	}

    option = {      # 
        name = me_scythia_01_greek_stance.8.a
        custom_tooltip = me_scythia_01_greek_stance.8.at
		current_ruler = { add_popularity = popularity_small  }
		reverse_add_opinion = {
			target = scope:c:OLB
			modifier = me_scy_01_diplo_largepos_opmod		### need to write
        }
		scope:me_scythia_scope = {
			add_alliance = {
				target = c:OLB
			}
		}
		hidden_effect = {
			capital_scope = {
				while = {
					count = 5
					limit = {
						num_of_slaves > 0
					}
					random_pops_in_province = {
						limit = {
							pop_type = slaves
						}
						kill_pop = yes
					}
				}
			}
			p:4529 = {
				while = {
					count = 5
					limit = {
						owner = OLB
					}
					define_pop = {
						type = slaves
						culture = scythian
						religion = indo_iranian_religion
					}
				}
			}
			add_treasury = 200
			scope:c:OLB = {
				add_treasury = -200
			}
		}
    }

	option = {      # 
		name = me_scythia_01_greek_stance.8.b
		custom_tooltip = me_scythia_01_greek_stance.8.bt
		current_ruler = { add_popularity = popularity_large  }
		add_treasury = -400
		reverse_add_opinion = {
			target = scope:c:OLB
			modifier = me_scy_01_diplo_smallpos_opmod		### need to write
		}
		scope:me_scythia_scope = {
			make_subject = {
				target = c:OLB
				type = client_state
			}
		}
		hidden_effect = {
			p:4529 = {
				add_building_level = commerce_building
				add_building_level = court_building
			}
			create_character = {
				add_finesse = 3
				add_charisma = 3
				add_zeal = 3
				add_trait = wise
				add_trait = scholar
				set_culture = bosporan
			}
		}
	}
