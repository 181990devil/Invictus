namespace = global_event

global_event.1 = { #Bactria Insult (Fire Condition)

    type = country_event
    hidden = yes

    trigger = {
        AND = {
            is_ai = no
            has_global_variable =  bactria_insult_sent
            NOT = { has_variable = bactria_insult_sent_fired }
        }
    }

    immediate = {
        trigger_event = {
            id = global_event.11
        }
    }

}

global_event.11 = { #Bactria Insult (Flavour)

	type = country_event
    title = global_event.11.t
    desc = global_event.11.desc
    picture = greeks_marching

    option = {
        name = global_event.11.a
        set_variable = { name = bactria_insult_sent_fired value = 1 days = -1 }
    }
}
