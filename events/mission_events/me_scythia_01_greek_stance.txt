	type = country_event

			}

    right_portrait = scope:c:SCY.current_ruler

			limit = { scope:me_scythia_1_scope.var:scythia_tyras_owner_offer = 1 }

				add_gold = 100

			limit = { scope:me_scythia_1_scope.var:scythia_tyras_owner_offer = 2 }

			current_ruler = {
				add_gold = 25
				add_corruption = 5
			}
		}
		else = {
			current_ruler = { 
				add_popularity = popularity_small  
			}
		}
	}

			set_variable = {
				name = scythia_tyras_owner_response
				value = 1
			}
		}

			target = scope:me_scythia_1_scope
			modifier = me_scythia_1_tyras_50_pos_opmod 
        }

			base = 1
			modifier = {
				factor = 2
				scope:me_scythia_1_scope = { is_ai = no }
			}
			modifier = {
				factor = 10
				stability <= 30
			}
			modifier = {
				factor = 5
				senate_approval <= 30
			}

				factor = 3
				calc_if_true = {
					limit = {
						current_ruler = {
							OR = {
								has_trait = weak_willed
								has_trait = corrupt
							}
						}
					}
				}
			}
			modifier = {
				factor = 5
				scope:me_scythia_1_scope.var:scythia_tyras_owner_offer = 1
			}
		}
	}
