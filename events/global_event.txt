global_event.1 = { #Bactria Insult (Fire Condition)

    type = country_event
    hidden = yes

    trigger = {
        AND = {
            has_global_variable =  bactria_insult_sent
            NOT = { has_variable = bactria_insult_sent_fired }
        }
    }

    immediate = {
        trigger_event = {
            id = global_event.2
        }
    }

}
