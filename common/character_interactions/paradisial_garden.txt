	on_other_nation = no
	on_own_nation = yes
	message = no
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_give_free_hands"
	
	potential_trigger = {
		hidden:scope:actor = {
			invention = paradisial_gardens_inv
			OR = {
				country_culture_group = persia
				any_integrated_culture = {
					has_culture_group = persia
					country_culture_pop_count >= 350
				}
			}
		}
		scope:target = {
			employer = scope:actor
			is_alive = yes
			NOT = { has_variable = olympic_attendee }
		}
	}

	allowed_trigger = {
		hidden:scope:actor = {
			treasury >= 150
			political_influence >= 15
		}
		hidden:scope:target = {
			NOT = {
				has_character_modifier = foreign_citizen
			}
			prisoner = no
			is_mercenary = no
			NOT = { has_trait = greedy }
			wealth >= 150
			any_holdings = {
				NOT = {
					has_province_modifier = {
						name = paradisial_gardens_modifier
					}
				}
			}
		}
	}
	
	effect = {
		scope:actor = {
			add_treasury = -150
			add_political_influecne = -15
			scope:target = {
				add_gold = -150
				random_holdings = {
					limit = {
						NOT = {
							has_province_modifier = paradisial_gardens_modifier
						}
					}
					add_province_modifier = {
						name = paradisial_gardens_modifier
						duration = 12775 #35 Years
					}
				}
			}
		}	
	}
	ai_will_do = {
		base = 1
		modifier = {
			add = 10
			scope:actor = {
				treasury >= 1000
			}
		}
		modifier = {
			add = 20
			scope:actor = {
				treasury >= 2000
			}
		}
		modifier = {
			add = -10
			scope:actor = {
				treasury <= 500
			}
		}
	}
	
}
