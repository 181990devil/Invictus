# First schemes activates the abilities to make character reform possible.


scheme_pyrrhonism = {
	
	can_be_picked = {
		scheme_preconditions_ruler = yes
		finesse > 3
		NOT = {
			has_character_modifier = aspiring_pyrrhonist_cmod
			has_character_modifier = pyrrhonist_cmod
		}
	}

	finished_when = {
			custom_tooltip = {
			text = "scheme_ambition_fulfillment_always_yes"
			custom_tooltip = { text = "" always = no }
		}
	}
	
	abort = {
		OR = {
			NOT = {
				has_variable = ongoing_scheme_target
			}
			is_alive = no
			is_ruler = no
			has_character_modifier = indisposed_charmod
		}
	}
	
	chance = {
		base = 1
	}
		
	duration = 90
	
	on_start = scheme_pyrrhonism_start
	on_finish = scheme_pyrrhonism_finish
	
}

scheme_modest = {
	
	can_be_picked = {
		scheme_preconditions_ruler = yes
		OR = {
			has_character_modifier = aspiring_pyrrhonist_cmod
			has_character_modifier = pyrrhonist_cmod
		}
		NOT = {
			has_trait = modest
		}
	}

			
	finished_when = {
		OR = {
			has_character_modifier = hold_games_modifier
			employer = {
			has_variable = summoned_curiate_assembly
			}
		}
	}
	
	abort = {
		OR = {
			NOT = {
				has_variable = ongoing_scheme_target
			}
			is_alive = no
			is_ruler = no
			has_character_modifier = indisposed_charmod
		}
	}
	
	chance = {
		base = 1
	}
		
	duration = 60
	
	on_start = scheme_modest_start
	on_finish = scheme_modest_finish
	
}
