gm_subject_simp = {
	icon = general_2
	header = mission_image_general

	repeatable = yes

	ai_chance = {
		base = 0

#		is_ai = no	# Caused a problem where it would need 1 day tick to become visible
		is_subject = yes
		has_civil_war = no

			has_variable = gm_subject_simp
		}

		save_scope_as = willing_subject

		}

			set_variable = {
				name = gm_subject_simp
				days = 7300	# 20 years
			}
		}
	}

	abort = {
		is_ai = yes
	}
	on_abort = {
		add_stability = -5
		hidden_effect = {
			set_variable = {
				name = gm_subject_simp
				days = 7300	# 20 years
			}
		}
	}
