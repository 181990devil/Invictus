academic_ambition = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_ruler = yes
			is_mercenary = no
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			employer = {
				any_character = {
					# has_any_office = yes
					has_character_modifier = academic_cmod
					has_character_modifier = scholarch
					# is_ruler = no
				}
			}
			NOR = {
			has_character_modifier = aspiring_epicurean_cmod
			has_character_modifier = aspiring_stoic_cmod
			has_character_modifier = aspiring_platonist_cmod
			has_character_modifier = aspiring_peripatetic_cmod
			has_character_modifier = aspiring_pyrrhonist_cmod
			has_character_modifier = aspiring_academic_cmod
			has_character_modifier = epicurean_cmod
			has_character_modifier = stoic_cmod
			has_character_modifier = platonist_cmod
			has_character_modifier = peripatetic_cmod
			has_character_modifier = pyrrhonist_cmod
			has_character_modifier = academic_cmod
			}
			# NOT = { has_trait = tolerant }
			NOT = { has_trait = lapsed }
		}
		
	}
	
	allowed_trigger = {
		scope:target = {
			NOT = {
				# OR = {
				has_ambition = scheme_academic
				# has_character_modifier = aspiring_platonist_cmod
				# }
			}
		}
	}
	
	effect = {
		scope:target = {
			clear_ambition_effect_ci = yes
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			}
			set_ambition = scheme_academic
		}
	}
	
}	




complete_academic_ambition = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_ruler = yes
			is_mercenary = no
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			has_character_modifier = aspiring_academic_cmod
		}
	}
	
	allowed_trigger = {
		scope:target = {
			has_trait = devout
			has_trait = righteous
			has_trait = sceptical
			NOT = {
				has_ambition = scheme_complete_academic_training
			}
		}
	}
	
	effect = {
		scope:target = {
			clear_ambition_effect_ci = yes
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			}
			set_ambition = scheme_complete_academic_training
		}
	}
}	


abandon_academic_ambition = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_ruler = yes
			is_mercenary = no
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			OR = {
				has_character_modifier = aspiring_academic_cmod
				has_character_modifier = academic_cmod
			}
		}
	}
	
	allowed_trigger = {
		scope:target = {
			NOT = {
				has_ambition = scheme_abandon_academic
			}
		}
	}
	
	effect = {
		scope:target = {
			clear_ambition_effect_ci = yes
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			}
			set_ambition = scheme_abandon_academic
		}
	}
	
}
